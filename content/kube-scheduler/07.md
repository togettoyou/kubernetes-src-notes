接上回，我们深入分析了调度框架（frameworkImpl）的实现，弄清楚了每个扩展点接口的语义，以及框架如何通过 Run* 方法驱动插件执行。本文将把目光移向调度器的主循环入口——`schedule_one.go`，看看一次完整的调度流程是怎样被串联起来的。

### 总览：一次调度的两个阶段

`schedule_one.go` 是整个调度流程的核心，其入口函数是 `scheduleOne`，它负责完成**单个 Pod 的完整调度工作**：

```go
// scheduleOne 完成单个 Pod 的全部调度流程，对调度算法的节点选择是串行的。
func (sched *Scheduler) scheduleOne(ctx context.Context) {
    // 从调度队列中取出下一个待调度的 Pod
    podInfo := sched.NextPod()
    if podInfo == nil || podInfo.Pod == nil {
        return
    }
    pod := podInfo.Pod

    // 根据 Pod 的 schedulerName 字段，找到对应的调度框架（Profile）
    fwk, err := sched.frameworkForPod(pod)
    if err != nil {
        klog.ErrorS(err, "Error occurred")
        return
    }
    // 跳过正在删除中的 Pod，或已经被 assume 过的 Pod
    if sched.skipPodSchedule(fwk, pod) {
        return
    }

    // 创建本轮调度的共享状态（CycleState）
    state := framework.NewCycleState()
    state.SetRecordPluginMetrics(rand.Intn(100) < pluginMetricsSamplePercent)
    podsToActivate := framework.NewPodsToActivate()
    state.Write(framework.PodsToActivateKey, podsToActivate)

    schedulingCycleCtx, cancel := context.WithCancel(ctx)
    defer cancel()

    // 第一阶段：调度周期（同步执行）——选出目标节点，并完成 Reserve、Permit
    scheduleResult, assumedPodInfo, status := sched.schedulingCycle(
        schedulingCycleCtx, state, fwk, podInfo, start, podsToActivate)
    if !status.IsSuccess() {
        sched.FailureHandler(schedulingCycleCtx, fwk, assumedPodInfo, status, scheduleResult.nominatingInfo, start)
        return
    }

    // 第二阶段：绑定周期（异步执行）——执行 PreBind、Bind、PostBind
    go func() {
        bindingCycleCtx, cancel := context.WithCancel(ctx)
        defer cancel()
        status := sched.bindingCycle(bindingCycleCtx, state, fwk, scheduleResult, assumedPodInfo, start, podsToActivate)
        if !status.IsSuccess() {
            sched.handleBindingCycleError(bindingCycleCtx, state, fwk, assumedPodInfo, start, scheduleResult, status)
        }
    }()
}
```

可以看到，整个流程被清晰地分成两个阶段：

- **调度周期（schedulingCycle）**：同步执行，负责从候选节点中选出一个最优节点，并完成 Reserve 和 Permit。这一阶段是串行的，每次只调度一个 Pod。
- **绑定周期（bindingCycle）**：异步执行（单独的 goroutine），负责将 Pod 真正绑定到节点。之所以异步，是为了不阻塞调度周期，让调度器能立刻开始下一个 Pod 的调度。

两个阶段之间通过 `assume`（乐观假设）机制解耦：调度周期选定节点后，立刻在调度缓存中乐观地记录这个 Pod 已经"跑在"目标节点上，从而让后续的调度能继续评估真实的资源状态，而无需等待实际绑定完成。

### 调度周期：schedulingCycle

调度周期的核心是 `schedulePod` 函数（对外暴露为 `sched.SchedulePod`），它承担了"从所有节点中找出最合适的那个"的任务：

```go
func (sched *Scheduler) schedulingCycle(
    ctx context.Context,
    state *framework.CycleState,
    fwk framework.Framework,
    podInfo *framework.QueuedPodInfo,
    start time.Time,
    podsToActivate *framework.PodsToActivate,
) (ScheduleResult, *framework.QueuedPodInfo, *framework.Status) {
    pod := podInfo.Pod

    // 核心：调用调度算法，选出目标节点
    scheduleResult, err := sched.SchedulePod(ctx, fwk, state, pod)
    if err != nil {
        // ... 如果所有节点都不满足，尝试运行 PostFilter（抢占）插件
        if !fwk.HasPostFilterPlugins() {
            return ScheduleResult{}, podInfo, framework.NewStatus(framework.Unschedulable).WithError(err)
        }
        result, status := fwk.RunPostFilterPlugins(ctx, state, pod, fitError.Diagnosis.NodeToStatusMap)
        // ...
        return ScheduleResult{nominatingInfo: nominatingInfo}, podInfo, framework.NewStatus(framework.Unschedulable).WithError(err)
    }

    // 乐观假设：在缓存中标记该 Pod 已绑定到目标节点
    assumedPodInfo := podInfo.DeepCopy()
    assumedPod := assumedPodInfo.Pod
    err = sched.assume(assumedPod, scheduleResult.SuggestedHost)
    if err != nil {
        return ScheduleResult{nominatingInfo: clearNominatedNode}, assumedPodInfo, framework.AsStatus(err)
    }

    // 运行 Reserve 插件，预留资源
    if sts := fwk.RunReservePluginsReserve(ctx, state, assumedPod, scheduleResult.SuggestedHost); !sts.IsSuccess() {
        // Reserve 失败，需要回滚：Unreserve + ForgetPod
        fwk.RunReservePluginsUnreserve(ctx, state, assumedPod, scheduleResult.SuggestedHost)
        sched.Cache.ForgetPod(assumedPod)
        return ScheduleResult{nominatingInfo: clearNominatedNode}, assumedPodInfo, sts
    }

    // 运行 Permit 插件，允许或延迟绑定
    runPermitStatus := fwk.RunPermitPlugins(ctx, state, assumedPod, scheduleResult.SuggestedHost)
    if !runPermitStatus.IsWait() && !runPermitStatus.IsSuccess() {
        // Permit 插件直接拒绝，同样需要回滚
        fwk.RunReservePluginsUnreserve(ctx, state, assumedPod, scheduleResult.SuggestedHost)
        sched.Cache.ForgetPod(assumedPod)
        return ScheduleResult{nominatingInfo: clearNominatedNode}, assumedPodInfo, runPermitStatus
    }

    // 调度周期结束时，激活由插件标记为可激活的 Pod
    if len(podsToActivate.Map) != 0 {
        sched.SchedulingQueue.Activate(podsToActivate.Map)
        podsToActivate.Map = make(map[string]*v1.Pod)
    }

    return scheduleResult, assumedPodInfo, nil
}
```

注意失败时的回滚顺序：Permit 失败 → Unreserve → ForgetPod。Reserve 失败同理。这与上一回介绍的 `RunReservePluginsUnreserve` 逆序回滚语义相对应，确保任何部分失败都能干净地撤销。

### schedulePod：选出目标节点

`schedulePod` 是实际执行节点选择的函数，它分三步完成工作：

```go
func (sched *Scheduler) schedulePod(ctx context.Context, fwk framework.Framework, state *framework.CycleState, pod *v1.Pod) (result ScheduleResult, err error) {
    // 第一步：刷新节点快照，确保调度基于最新的集群状态
    if err := sched.Cache.UpdateSnapshot(sched.nodeInfoSnapshot); err != nil {
        return result, err
    }
    if sched.nodeInfoSnapshot.NumNodes() == 0 {
        return result, ErrNoNodesAvailable
    }

    // 第二步：过滤——找出所有满足 Pod 要求的候选节点
    feasibleNodes, diagnosis, err := sched.findNodesThatFitPod(ctx, fwk, state, pod)
    if err != nil {
        return result, err
    }
    if len(feasibleNodes) == 0 {
        return result, &framework.FitError{Pod: pod, NumAllNodes: sched.nodeInfoSnapshot.NumNodes(), Diagnosis: diagnosis}
    }

    // 剪枝：如果只有一个候选节点，直接使用，无需打分
    if len(feasibleNodes) == 1 {
        return ScheduleResult{SuggestedHost: feasibleNodes[0].Name, ...}, nil
    }

    // 第三步：打分——对候选节点按优先级排序，选出最优节点
    priorityList, err := prioritizeNodes(ctx, sched.Extenders, fwk, state, pod, feasibleNodes)
    if err != nil {
        return result, err
    }
    host, err := selectHost(priorityList)

    return ScheduleResult{SuggestedHost: host, ...}, err
}
```

流程清晰：快照 → 过滤 → 打分 → 选主机。如果过滤后只剩一个节点，则直接省略打分步骤，这是一个小而实用的优化。

### 过滤阶段：findNodesThatFitPod

`findNodesThatFitPod` 负责从所有节点中筛选出满足 Pod 调度条件的候选节点，它把整个过滤过程分成几个步骤：

```go
func (sched *Scheduler) findNodesThatFitPod(ctx context.Context, fwk framework.Framework, state *framework.CycleState, pod *v1.Pod) ([]*v1.Node, framework.Diagnosis, error) {
    diagnosis := framework.Diagnosis{
        NodeToStatusMap:      make(framework.NodeToStatusMap),
        UnschedulablePlugins: sets.NewString(),
    }

    allNodes, _ := sched.nodeInfoSnapshot.NodeInfos().List()

    // 步骤一：运行 PreFilter 插件，预处理并收窄候选节点集合
    preRes, s := fwk.RunPreFilterPlugins(ctx, state, pod)
    if !s.IsSuccess() {
        // PreFilter 直接失败，记录原因并返回
        diagnosis.PreFilterMsg = s.Message()
        return nil, diagnosis, nil
    }

    // 步骤二：优先评估被提名节点（来自上一轮抢占的结果）
    if len(pod.Status.NominatedNodeName) > 0 {
        feasibleNodes, err := sched.evaluateNominatedNode(ctx, pod, fwk, state, diagnosis)
        if len(feasibleNodes) != 0 {
            return feasibleNodes, diagnosis, nil // 被提名节点通过，直接返回
        }
    }

    // 步骤三：根据 PreFilter 的结果，确定参与评估的节点范围
    nodes := allNodes
    if !preRes.AllNodes() {
        // PreFilter 插件缩小了范围，只评估指定节点
        nodes = /* 从 preRes.NodeNames 中获取 NodeInfo 列表 */
    }

    // 步骤四：并行过滤——运行所有 Filter 插件
    feasibleNodes, err := sched.findNodesThatPassFilters(ctx, fwk, state, pod, diagnosis, nodes)
    // 更新下一轮过滤的起始节点索引，保证每个节点被公平遍历
    processedNodes := len(feasibleNodes) + len(diagnosis.NodeToStatusMap)
    sched.nextStartNodeIndex = (sched.nextStartNodeIndex + processedNodes) % len(nodes)

    // 步骤五：通过 Extender（外部扩展器）进行额外过滤
    feasibleNodes, err = findNodesThatPassExtenders(sched.Extenders, pod, feasibleNodes, diagnosis.NodeToStatusMap)
    return feasibleNodes, diagnosis, err
}
```

这里有几个值得关注的设计细节：

**被提名节点的优先处理**：如果 Pod 的 `status.nominatedNodeName` 不为空（说明在上一轮调度中通过抢占确定了目标节点），调度器会优先检查这个节点。如果它仍然满足条件，直接用它，无需再遍历所有节点，大幅缩短调度耗时。

**nextStartNodeIndex 的轮转**：过滤不一定遍历所有节点（后面会讲到 `numFeasibleNodesToFind` 的截断逻辑），为了保证每个节点都有被选中的机会，调度器维护了一个 `nextStartNodeIndex`，每轮过滤后向前推进，形成类似"环形遍历"的效果。

### 并行过滤：findNodesThatPassFilters

过滤阶段对所有节点并行执行 Filter 插件，这是调度器性能的关键所在：

```go
func (sched *Scheduler) findNodesThatPassFilters(
    ctx context.Context, fwk framework.Framework,
    state *framework.CycleState, pod *v1.Pod,
    diagnosis framework.Diagnosis, nodes []*framework.NodeInfo,
) ([]*v1.Node, error) {
    numAllNodes := len(nodes)
    // 根据集群规模计算本轮需要找到的可行节点数（不一定遍历全部节点）
    numNodesToFind := sched.numFeasibleNodesToFind(fwk.PercentageOfNodesToScore(), int32(numAllNodes))

    feasibleNodes := make([]*v1.Node, numNodesToFind)
    var feasibleNodesLen int32
    ctx, cancel := context.WithCancel(ctx)
    defer cancel()

    checkNode := func(i int) {
        // 从 nextStartNodeIndex 开始，轮转遍历节点列表
        nodeInfo := nodes[(sched.nextStartNodeIndex+i)%numAllNodes]
        // 考虑被提名 Pod 的资源占用，运行 Filter 插件（含两轮检查）
        status := fwk.RunFilterPluginsWithNominatedPods(ctx, state, pod, nodeInfo)
        if status.IsSuccess() {
            length := atomic.AddInt32(&feasibleNodesLen, 1)
            if length > numNodesToFind {
                // 已找到足够多的可行节点，取消其他 goroutine
                cancel()
                atomic.AddInt32(&feasibleNodesLen, -1)
            } else {
                feasibleNodes[length-1] = nodeInfo.Node()
            }
        } else {
            // 过滤失败，记录失败原因
            statusesLock.Lock()
            diagnosis.NodeToStatusMap[nodeInfo.Node().Name] = status
            diagnosis.UnschedulablePlugins.Insert(status.FailedPlugin())
            statusesLock.Unlock()
        }
    }

    // 并行执行，默认并发度为 16
    fwk.Parallelizer().Until(ctx, numAllNodes, checkNode, metrics.Filter)
    return feasibleNodes[:feasibleNodesLen], nil
}
```

找到足够多的可行节点后，通过 `cancel()` 取消其他正在进行的检查，这是一个典型的"提前终止"优化。

#### 可行节点数的计算：numFeasibleNodesToFind

对于大规模集群，逐一评估所有节点的成本很高。`numFeasibleNodesToFind` 实现了一个自适应的截断策略：

```go
func (sched *Scheduler) numFeasibleNodesToFind(percentageOfNodesToScore *int32, numAllNodes int32) (numNodes int32) {
    // 集群节点数少于 100 时，全量评估
    if numAllNodes < minFeasibleNodesToFind {
        return numAllNodes
    }

    var percentage int32
    if percentageOfNodesToScore != nil {
        percentage = *percentageOfNodesToScore // 优先使用 Profile 中配置的比例
    } else {
        percentage = sched.percentageOfNodesToScore // 使用全局配置
    }

    if percentage == 0 {
        // 未配置比例时，使用自适应公式：集群越大，比例越低（最低 5%）
        percentage = int32(50) - numAllNodes/125
        if percentage < minFeasibleNodesPercentageToFind {
            percentage = minFeasibleNodesPercentageToFind // 最低 5%
        }
    }

    numNodes = numAllNodes * percentage / 100
    if numNodes < minFeasibleNodesToFind {
        return minFeasibleNodesToFind // 至少 100 个
    }
    return numNodes
}
```

这个公式体现了一个平衡：节点越多，全量过滤的成本越高，可以容忍一定的"近似最优"来换取性能。例如，1000 个节点的集群，默认会评估约 `50 - 1000/125 = 42%` 的节点（即约 420 个）；10000 个节点的集群则约评估 `50 - 80 = 5%`（即 500 个，由最低阈值保底）。

### 打分阶段：prioritizeNodes 与 selectHost

过滤出候选节点后，进入打分阶段：

```go
func prioritizeNodes(
    ctx context.Context,
    extenders []framework.Extender,
    fwk framework.Framework,
    state *framework.CycleState,
    pod *v1.Pod,
    nodes []*v1.Node,
) ([]framework.NodePluginScores, error) {
    // 如果没有任何打分插件也没有 Extender，所有节点得分均为 1
    if len(extenders) == 0 && !fwk.HasScorePlugins() {
        result := make([]framework.NodePluginScores, 0, len(nodes))
        for i := range nodes {
            result = append(result, framework.NodePluginScores{Name: nodes[i].Name, TotalScore: 1})
        }
        return result, nil
    }

    // 步骤一：运行 PreScore 插件
    preScoreStatus := fwk.RunPreScorePlugins(ctx, state, pod, nodes)
    if !preScoreStatus.IsSuccess() {
        return nil, preScoreStatus.AsError()
    }

    // 步骤二：运行 Score 插件（内部包含三步并行：打分 → 归一化 → 加权汇总）
    nodesScores, scoreStatus := fwk.RunScorePlugins(ctx, state, pod, nodes)
    if !scoreStatus.IsSuccess() {
        return nil, scoreStatus.AsError()
    }

    // 步骤三：如果有 Extender，并行运行 Extender 的打分逻辑，并合并到总分中
    if len(extenders) != 0 && nodes != nil {
        var mu sync.Mutex
        var wg sync.WaitGroup
        for i := range extenders {
            if !extenders[i].IsInterested(pod) {
                continue
            }
            wg.Add(1)
            go func(extIndex int) {
                defer wg.Done()
                prioritizedList, weight, err := extenders[extIndex].Prioritize(pod, nodes)
                // 将 Extender 分数缩放到 [0, MaxNodeScore] 范围，然后合并
                // ...
            }(i)
        }
        wg.Wait()
        // 将 Extender 分数追加到 nodesScores 中
    }

    return nodesScores, nil
}
```

Extender 的打分是并行的（多个 goroutine 同时调用多个 Extender），但需要注意的是，Extender 分数需要缩放到与内置插件相同的范围（`[0, MaxNodeScore]`，即 `[0, 100]`）才能合并。

打分完成后，`selectHost` 从总分最高的节点中选出一个：

```go
// selectHost 采用蓄水池抽样算法，从最高分的节点中随机选一个（处理分数相同的情况）
func selectHost(nodeScores []framework.NodePluginScores) (string, error) {
    if len(nodeScores) == 0 {
        return "", fmt.Errorf("empty priorityList")
    }
    maxScore := nodeScores[0].TotalScore
    selected := nodeScores[0].Name
    cntOfMaxScore := 1
    for _, ns := range nodeScores[1:] {
        if ns.TotalScore > maxScore {
            maxScore = ns.TotalScore
            selected = ns.Name
            cntOfMaxScore = 1
        } else if ns.TotalScore == maxScore {
            cntOfMaxScore++
            // 以 1/cntOfMaxScore 的概率替换当前候选节点
            if rand.Intn(cntOfMaxScore) == 0 {
                selected = ns.Name
            }
        }
    }
    return selected, nil
}
```

当多个节点总分相同时，简单选第一个会有偏向性（总是选同一个节点，导致负载不均衡）。`selectHost` 采用**蓄水池抽样（Reservoir Sampling）**，保证所有同分节点被选中的概率相等，有助于在同等候选节点间实现均匀分布。

### assume：乐观假设机制

选出目标节点后，调度周期并不等待实际绑定完成，而是立即在缓存中"假设"这个 Pod 已经运行在目标节点上：

```go
// assume 将 Pod 的 NodeName 设置为目标节点，并在调度缓存中乐观地标记其为"已假设"状态
func (sched *Scheduler) assume(assumed *v1.Pod, host string) error {
    // 直接修改 Pod 的 NodeName 字段
    assumed.Spec.NodeName = host

    if err := sched.Cache.AssumePod(assumed); err != nil {
        klog.ErrorS(err, "Scheduler cache AssumePod failed")
        return err
    }
    // 如果该 Pod 曾经是被提名 Pod，现在已经有了确定的节点，删除提名信息
    if sched.SchedulingQueue != nil {
        sched.SchedulingQueue.DeleteNominatedPodIfExists(assumed)
    }
    return nil
}
```

这个"假设"的有效期很短——从 assume 到实际绑定成功（`Cache.FinishBinding`）或失败（`Cache.ForgetPod`）之间的时间窗口。在此期间，后续 Pod 的调度会看到这个"假设中的 Pod"占据了目标节点的资源，从而避免双重调度。

### 绑定周期：bindingCycle

绑定周期在独立 goroutine 中异步执行：

```go
func (sched *Scheduler) bindingCycle(
    ctx context.Context,
    state *framework.CycleState,
    fwk framework.Framework,
    scheduleResult ScheduleResult,
    assumedPodInfo *framework.QueuedPodInfo,
    start time.Time,
    podsToActivate *framework.PodsToActivate,
) *framework.Status {
    assumedPod := assumedPodInfo.Pod

    // 第一步：等待 Permit 插件的放行信号（如果调度周期中有插件返回了 Wait）
    if status := fwk.WaitOnPermit(ctx, assumedPod); !status.IsSuccess() {
        return status
    }

    // 第二步：运行 PreBind 插件（如 volumebinding 完成 PVC 绑定）
    if status := fwk.RunPreBindPlugins(ctx, state, assumedPod, scheduleResult.SuggestedHost); !status.IsSuccess() {
        return status
    }

    // 第三步：执行绑定（先尝试 Extender，再调用 Bind 插件）
    if status := sched.bind(ctx, fwk, assumedPod, scheduleResult.SuggestedHost, state); !status.IsSuccess() {
        return status
    }

    klog.V(2).InfoS("Successfully bound pod to node", "pod", klog.KObj(assumedPod), "node", scheduleResult.SuggestedHost, ...)
    metrics.PodScheduled(fwk.ProfileName(), metrics.SinceInSeconds(start))

    // 第四步：运行 PostBind 插件（通知性，失败不影响绑定结果）
    fwk.RunPostBindPlugins(ctx, state, assumedPod, scheduleResult.SuggestedHost)

    // 绑定周期结束时，激活由插件标记的 Pod
    if len(podsToActivate.Map) != 0 {
        sched.SchedulingQueue.Activate(podsToActivate.Map)
    }
    return nil
}
```

注意绑定周期的第一步是 `WaitOnPermit`，它会阻塞直到 Permit 插件给出最终信号（Allow 或 Reject）。这就是调度周期和绑定周期之间的唯一同步点——调度周期的 `RunPermitPlugins` 启动等待，绑定周期的 `WaitOnPermit` 阻塞等待放行。

`bind` 方法遵循"Extender 优先"原则：

```go
func (sched *Scheduler) bind(ctx context.Context, fwk framework.Framework, assumed *v1.Pod, targetNode string, state *framework.CycleState) (status *framework.Status) {
    defer func() {
        // 无论成功还是失败，都通知缓存绑定已完成（清除 assumed 标记）
        sched.finishBinding(fwk, assumed, targetNode, status)
    }()

    // 先尝试 Extender 绑定
    bound, err := sched.extendersBinding(assumed, targetNode)
    if bound {
        return framework.AsStatus(err) // Extender 处理了绑定，直接返回
    }
    // 再调用框架的 Bind 插件（通常是 DefaultBinder）
    return fwk.RunBindPlugins(ctx, state, assumed, targetNode)
}
```

绑定完成后，`finishBinding` 会调用 `Cache.FinishBinding` 来结束 Pod 的 assumed 状态，并通过 `EventRecorder` 向 API Server 写入一个 `Scheduled` 事件。

### 绑定失败的处理：handleBindingCycleError

如果绑定周期任意环节失败，需要进行完整的回滚：

```go
func (sched *Scheduler) handleBindingCycleError(
    ctx context.Context, state *framework.CycleState, fwk framework.Framework,
    podInfo *framework.QueuedPodInfo, start time.Time,
    scheduleResult ScheduleResult, status *framework.Status) {

    assumedPod := podInfo.Pod
    // 运行 Unreserve 插件，回滚 Reserve 阶段的状态
    fwk.RunReservePluginsUnreserve(ctx, state, assumedPod, scheduleResult.SuggestedHost)

    // 从调度缓存中移除这个 assumed Pod，释放占用的资源
    if forgetErr := sched.Cache.ForgetPod(assumedPod); forgetErr != nil {
        klog.ErrorS(forgetErr, "scheduler cache ForgetPod failed")
    } else {
        // ForgetPod 成功后，触发一次 AssignedPodDelete 事件，尝试唤醒其他不可调度的 Pod
        // （这里用 defer 是为了避免把当前 Pod 本身也移回队列）
        if status.IsUnschedulable() {
            defer sched.SchedulingQueue.MoveAllToActiveOrBackoffQueue(internalqueue.AssignedPodDelete, func(pod *v1.Pod) bool {
                return assumedPod.UID != pod.UID // 排除自身
            })
        } else {
            sched.SchedulingQueue.MoveAllToActiveOrBackoffQueue(internalqueue.AssignedPodDelete, nil)
        }
    }

    // 最终由 FailureHandler 将 Pod 重新放回调度队列
    sched.FailureHandler(ctx, fwk, podInfo, status, clearNominatedNode, start)
}
```

值得注意的是对 `Unschedulable` 状态的特殊处理：绑定失败时，已被 assume 的 Pod 占用的资源被释放，这本应触发其他 Pod 的重新调度；但如果失败原因是 `Unschedulable`（例如 Permit 插件拒绝），把当前 Pod 自身也移回活动队列没有意义，因此用 `defer` 和过滤函数将其排除在外。

### 总结

本文完整梳理了 `schedule_one.go` 中一次调度的全链路：

- **scheduleOne** 是入口，将调度分为同步的**调度周期**和异步的**绑定周期**两个阶段。
- **schedulingCycle** → **schedulePod** → **findNodesThatFitPod** → **findNodesThatPassFilters**：层层调用，完成从快照刷新、PreFilter、Filter 到节点筛选的全过程。
- **numFeasibleNodesToFind** 实现了大集群下的自适应截断，避免对所有节点逐一评估。
- **prioritizeNodes** → **selectHost**：对候选节点打分后，通过蓄水池抽样选出目标节点，保证同分节点的均匀选择。
- **assume** 机制解耦了调度与绑定，是调度器高吞吐的关键设计。
- **bindingCycle** 异步执行 WaitOnPermit → PreBind → Bind → PostBind，任一失败都会触发完整的回滚（Unreserve + ForgetPod）。

至此，kube-scheduler 的核心调度流程就完整地呈现出来了。
